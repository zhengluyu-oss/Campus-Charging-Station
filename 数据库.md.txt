# 3.订单表
CREATE TABLE orders (
                        order_id INT PRIMARY KEY AUTO_INCREMENT,
                        user_id int NOT NULL COMMENT '用户名称',
                        station_id int NOT NULL COMMENT '充电桩名称',
                        start_time TIMESTAMP NOT NULL COMMENT '开始时间',
                        end_time TIMESTAMP COMMENT '结束时间',
                        duration_minutes INT COMMENT '持续分钟数',
                        total_amount DECIMAL(10,2) COMMENT '总金额',
                        payment_status ENUM('pending', 'paid', 'failed') DEFAULT 'pending' COMMENT '支付状态',
                        order_status ENUM('active', 'completed', 'cancelled') DEFAULT 'active' COMMENT '订单状态',
                        created_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    -- 外键约束
                        FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
                        FOREIGN KEY (station_id) REFERENCES charging_stations(station_id) ON DELETE CASCADE
);



## 4. 支付记录表 (payments)
CREATE TABLE payments (
                          payment_id INT PRIMARY KEY AUTO_INCREMENT,
                          order_id INT NOT NULL COMMENT '订单ID',
                          amount DECIMAL(10,2) NOT NULL COMMENT '金额',
                          payments_method ENUM('cash', 'creditCard', 'alipay', 'wechatPpay') NOT NULL COMMENT '支付方式',
                          payments_status ENUM('success', 'failed', 'refunded') COMMENT '支付状态',  # 支付状态  成功  失败 已退款
                          paid_time TIMESTAMP COMMENT '支付时间',

    -- 外键约束
                          FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE
);



## 5. 预约表 (reservations)
CREATE TABLE reservations (
                              reservation_id INT PRIMARY KEY AUTO_INCREMENT,
                              user_id INT NOT NULL COMMENT '用户ID',
                              station_id INT NOT NULL COMMENT '充电桩ID',
                              reserved_start_time TIMESTAMP NOT NULL COMMENT '预约开始时间',
                              reserved_end_time TIMESTAMP NOT NULL COMMENT '预约结束时间',
                              status varchar(255) COMMENT '预约状态',
                              created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    -- 外键约束
                              FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
                              FOREIGN KEY (station_id) REFERENCES charging_stations(station_id) ON DELETE CASCADE
);




## 6. 系统日志表 (system_logs)(记录用户的充电预约记录)
CREATE TABLE system_logs (
                             log_id INT PRIMARY KEY AUTO_INCREMENT,
                             user_id INT COMMENT '用户ID',
                             action VARCHAR(100) NOT NULL COMMENT '操作',
                             details TEXT COMMENT '详情',
                             ip_address VARCHAR(45) COMMENT 'IP地址',
                             created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    -- 外键约束
                             FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL
);

# 8. 积分表格（用户每冲1分钟电获得0.05积分）
create table user_points (
                             id int primary key auto_increment,
                             user_id int not null comment '用户id',
                             points int not null default 0 comment '积分数',
                             created_time datetime not null default current_timestamp comment '创建时间',
                             updated_time datetime not null default current_timestamp on update current_timestamp comment '更新时间',
                             foreign key (user_id) references users(user_id)
);



# 9.积分变动表（记录用户获得积分和消费积分）
CREATE TABLE points_record (
                               id BIGINT PRIMARY KEY AUTO_INCREMENT,
                               user_id BIGINT NOT NULL COMMENT '用户ID',
                               points_change INT NOT NULL COMMENT '积分变动值（正为增加，负为减少）',
                               balance_after INT NOT NULL COMMENT '变动后总积分',
                               reference_id VARCHAR(64) DEFAULT NULL COMMENT '关联业务ID（如订单号）',
                               created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

                               KEY idx_user_id (user_id),
                               KEY idx_created_at (created_at)
) COMMENT '积分变动记录表';




# 10.卡卷表格,记录卡卷信息（这是商品的卡卷，可以在商城中使用）
CREATE TABLE coupon_template (
                                 id BIGINT PRIMARY KEY,
                                 name VARCHAR(100) NOT NULL COMMENT '券名称”',
                                 type TINYINT NOT NULL COMMENT '1-满减券, 2-折扣券, 3-代金券',
                                 discount_value DECIMAL(10,2) NOT NULL,
                                 min_order_amount DECIMAL(10,2),
                                 valid_days INT COMMENT '领取后有效天数',
                                 total_count INT DEFAULT -1,
                                 per_user_limit INT DEFAULT 1,
                                 created_at DATETIME
);


# 11.用户卡券记录表（记录谁领了哪张券）
CREATE TABLE user_coupon (
                             id BIGINT PRIMARY KEY,
                             user_id BIGINT NOT NULL,
                             coupon_id BIGINT NOT NULL COMMENT '关联 coupon_template.id',
                             status TINYINT DEFAULT 0 COMMENT '0-未使用, 1-已使用, 2-已过期',
                             received_at DATETIME,
                             used_at DATETIME,

# 13.商城订单表
CREATE TABLE `goods_orders` (
                                `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '订单ID（主键）',

    -- 用户信息
                                `user_id` BIGINT NOT NULL COMMENT '下单用户ID',
                                `username` VARCHAR(50) NOT NULL COMMENT '下单时用户名（冗余，防用户改名）',

    -- 订单基础信息
                                `order_no` VARCHAR(32) NOT NULL COMMENT '订单编号（唯一，如 ORD20260201123456）',
                                `order_status` TINYINT NOT NULL DEFAULT 0 COMMENT '订单状态：0-待支付, 1-已支付, 2-已发货, 3-已完成, 4-已取消, 5-退款中, 6-已退款',
                                `order_type` TINYINT NOT NULL DEFAULT 1 COMMENT '订单类型：1-普通商品, 2-秒杀, 3-团购',

    -- 金额相关（单位：分，避免浮点误差）
                                `total_amount` INT NOT NULL COMMENT '订单总金额（分）',
                                `pay_amount` INT NOT NULL COMMENT '实付金额（分）',
                                `discount_amount` INT NOT NULL DEFAULT 0 COMMENT '优惠金额（分）',
                                `freight_amount` INT NOT NULL DEFAULT 0 COMMENT '运费（分）',

    -- 支付信息
                                `pay_type` TINYINT DEFAULT NULL COMMENT '支付方式：1-微信, 2-支付宝, 3-余额',
                                `pay_time` DATETIME DEFAULT NULL COMMENT '支付时间',
                                `transaction_id` VARCHAR(64) DEFAULT NULL COMMENT '第三方支付流水号',

    -- 时间信息
                                `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
                                `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
                                `confirm_time` DATETIME DEFAULT NULL COMMENT '确认收货时间',

    -- 其他
                                `remark` VARCHAR(200) DEFAULT NULL COMMENT '用户备注',
                                `source` TINYINT NOT NULL DEFAULT 1 COMMENT '下单来源：1-PC, 2-H5, 3-小程序, 4-APP',

                                PRIMARY KEY (`id`),
                                UNIQUE KEY `uk_order_no` (`order_no`),
                                KEY `idx_user_id` (`user_id`),
                                KEY `idx_create_time` (`create_time`),
                                KEY `idx_order_status` (`order_status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT ='商城订单主表';

# 15.红包类型表
CREATE TABLE red_packet (
                            id INT PRIMARY KEY AUTO_INCREMENT,
                            name VARCHAR(50) NOT NULL COMMENT '红包名称',
                            amount DECIMAL(10,2) NOT NULL COMMENT '面值（元）',
                            min_charge DECIMAL(10,2) DEFAULT NULL COMMENT '使用门槛（充电金额满多少可用）',
                            total_count INT NOT NULL DEFAULT -1 COMMENT '总数量，-1表示不限量',
                            per_user_limit INT NOT NULL DEFAULT 1 COMMENT '每人限领数量',
                            valid_days INT NOT NULL DEFAULT 7 COMMENT '领取后有效天数',
                            status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：1-启用，2-停用'
);


-- 16. 用户红包表
CREATE TABLE user_red_packet (
                                 id INT PRIMARY KEY AUTO_INCREMENT,
                                 user_id INT NOT NULL COMMENT '用户ID',
                                 packet_id INT NOT NULL COMMENT '红包模板ID',
                                 status TINYINT NOT NULL DEFAULT 0 COMMENT '状态：0-未使用，1-已使用，2-已过期',
                                 received_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '领取时间',
                                 used_time DATETIME DEFAULT NULL COMMENT '使用时间',
                                 expired_time DATETIME NOT NULL COMMENT '过期时间',

                                 FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
                                 FOREIGN KEY (packet_id) REFERENCES red_packet(id) ON DELETE CASCADE
);
